CREATE VIEW calibration_curve AS
SELECT
  run_id,
  confidence_bin,
  COUNT(*) as n,
  ROUND(AVG(predicted_confidence), 3) as avg_predicted,
  ROUND(AVG(CAST(realized_outcome AS REAL)), 3) as avg_actual,
  ROUND(AVG((predicted_confidence - CAST(realized_outcome AS REAL)) * (predicted_confidence - CAST(realized_outcome AS REAL))), 4) as bin_brier
FROM backtest_results
GROUP BY run_id, confidence_bin
ORDER BY run_id, confidence_bin
CREATE VIEW outcome_summary AS
SELECT
  action,
  COUNT(*) as count,
  ROUND(AVG(json_extract(realized, '$.stars_delta')), 1) as avg_stars,
  ROUND(AVG(json_extract(realized, '$.downloads_delta')), 1) as avg_downloads
FROM outcomes
WHERE action IS NOT NULL
GROUP BY action
CREATE VIEW shortlist AS
SELECT
  p.id,
  p.title,
  p.url,
  p.source,
  p.score,
  p.confidence,
  p.status,
  p.created_at,
  p.updated_at,
  MAX(CASE WHEN s.signal_type = 'pain_intensity' THEN s.value END) as pain_intensity,
  MAX(CASE WHEN s.signal_type = 'monetizability' THEN s.value END) as monetizability,
  MAX(CASE WHEN s.signal_type = 'competitor_gap' THEN s.value END) as competitor_gap,
  MAX(CASE WHEN s.signal_type = 'evidence_quality' THEN s.value END) as evidence_quality
FROM projects p
LEFT JOIN signals s ON p.id = s.project_id
WHERE p.score >= 60 AND p.confidence >= 0.6
GROUP BY p.id
ORDER BY p.score DESC
CREATE VIEW shortlisted AS
SELECT * FROM projects
WHERE status = 'scored' AND score >= 60
ORDER BY score DESC
CREATE VIEW source_stats AS
SELECT
  source,
  COUNT(*) as total_projects,
  ROUND(AVG(score), 1) as avg_score,
  SUM(CASE WHEN score >= 60 THEN 1 ELSE 0 END) as shortlisted_count,
  MAX(updated_at) as last_update
FROM projects
GROUP BY source
